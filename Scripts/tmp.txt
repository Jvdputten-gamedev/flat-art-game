func _on_adjacent_pressed():
	clear_terrain()
	var cells = AttackPatterns.get_all_adjacent(from=center_cell)
	paint_hex_on_map(cells)


func _on_triangle_up_pressed():
	clear_terrain()
	var cells = AttackPatterns.get_triangle_up(from=center_cell)
	paint_hex_on_map(cells)


func _on_triangle_down_pressed():
	clear_terrain()
	var cells = AttackPatterns.get_triangle_down(from=center_cell)
	paint_hex_on_map(cells)


func _on_direction_item_selected(index):
	clear_terrain()
	var cells = Array()
	var dir = Directions.DIRECTIONS[index]
	cells.append(AttackPatterns.get_adjacent(from=center_cell, dir=dir.cube_coords).oddq_coords)
	paint_hex_on_map(cells)
